        if !s.is_empty() {
            let parts: Vec<serde_json::Value> = s.split('|')
                .map(|p| serde_json::Value::String(p.trim().to_string()))
                .filter(|v| !v.as_str().map(|s| s.is_empty()).unwrap_or(true))
                .collect();
            uso.insert(k.clone(), serde_json::Value::Array(parts));

}
